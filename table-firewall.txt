CREATE TABLE FIREWALL
(
  ID            NUMBER(11)                      NOT NULL,
  DB            VARCHAR2(50 BYTE)               NOT NULL,
  USER_SESSION  VARCHAR2(50 BYTE),
  IP_ADDRESS    VARCHAR2(50 BYTE),
  MODULE        VARCHAR2(50 BYTE),
  ID_FECHA      DATE
)
TABLESPACE users
PCTFREE    10
INITRANS   1
MAXTRANS   255
STORAGE    (
            INITIAL          64K
            NEXT             1M
            MINEXTENTS       1
            MAXEXTENTS       UNLIMITED
            PCTINCREASE      0
            BUFFER_POOL      DEFAULT
           )
LOGGING 
NOCOMPRESS 
NOCACHE;


DROP SEQUENCE FIREWALL_SEQ;

CREATE SEQUENCE FIREWALL_SEQ
  START WITH 1
  MAXVALUE 999999999999999999999999999
  MINVALUE 1
  NOCYCLE
  CACHE 20
  NOORDER
  NOKEEP
  NOSCALE
  GLOBAL;


CREATE OR REPLACE TRIGGER FIREWALL_TRG
BEFORE INSERT
ON FIREWALL
REFERENCING NEW AS New OLD AS Old
FOR EACH ROW
BEGIN
-- For Toad:  Highlight column ID
  :new.ID := FIREWALL_SEQ.nextval;
END FIREWALL_TRG;
/
